FROM mcr.microsoft.com/dotnet/aspnet:8.0 AS base
WORKDIR /app
EXPOSE 8080
EXPOSE 8081

FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
ARG BUILD_CONFIGURATION=Release
WORKDIR /src

# Copy project files
COPY ["src/Services/OrderManagement.OrderService/OrderManagement.OrderService.csproj", "src/Services/OrderManagement.OrderService/"]
COPY ["src/Common/OrderManagement.Common/OrderManagement.Common.csproj", "src/Common/OrderManagement.Common/"]

# Restore dependencies
RUN dotnet restore "src/Services/OrderManagement.OrderService/OrderManagement.OrderService.csproj"

# Copy all source code
COPY . .

# Build the application
WORKDIR "/src/src/Services/OrderManagement.OrderService"
RUN dotnet build "OrderManagement.OrderService.csproj" -c $BUILD_CONFIGURATION -o /app/build

FROM build AS publish
ARG BUILD_CONFIGURATION=Release
RUN dotnet publish "OrderManagement.OrderService.csproj" -c $BUILD_CONFIGURATION -o /app/publish /p:UseAppHost=false

FROM base AS final
WORKDIR /app
COPY --from=publish /app/publish .

# Create logs directory
RUN mkdir -p /app/logs

ENTRYPOINT ["dotnet", "OrderManagement.OrderService.dll"]
